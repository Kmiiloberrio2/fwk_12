create procedure Contactos_g	
(	
	
	
	 @Telefonos       varchar(500) 
	

	)
	
	AS
	declare @CurrentDate datetime
	set @CurrentDate  = getdate()
	select  IDLead,Fecha from Contactos where  Telefonos = @Telefonos AND
	DATEDIFF(minute,Fecha,@CurrentDate) <40
